38   def heatmap(data, row_labels, col_labels, ax=None,
39               cbar_kw=None, cbarlabel="", **kwargs):
40       """ 
41       Create a heatmap from a numpy array and two lists of labels. 
42    
43       Parameters 
44       ---------- 
45       data 
46           A 2D numpy array of shape (M, N). 
47       row_labels 
48           A list or array of length M with the labels for the rows. 
49       col_labels 
50           A list or array of length N with the labels for the columns. 
51       ax 
52           A `matplotlib.axes.Axes` instance to which the heatmap is plotted.  If 
53           not provided, use current axes or create a new one.  Optional. 
54       cbar_kw 
55           A dictionary with arguments to `matplotlib.Figure.colorbar`.  Optional. 
56       cbarlabel 
57           The label for the colorbar.  Optional. 
58       **kwargs 
59           All other arguments are forwarded to `imshow`. 
60       """
61   
62       if ax is None:
63           ax = plt.gca()
64   
65       if cbar_kw is None:
66           cbar_kw = {}
67   
68       # Plot the heatmap
69       im = ax.imshow(data, **kwargs)
70   
71       # Create colorbar
72       cbar = ax.figure.colorbar(im, ax=ax, **cbar_kw)
73       cbar.ax.set_ylabel(cbarlabel, rotation=-90, va="bottom")
74   
75       # Show all ticks and label them with the respective list entries.
76       ax.set_xticks(np.arange(data.shape[1]), labels=col_labels)
77       ax.set_yticks(np.arange(data.shape[0]), labels=row_labels)
78   
79       # Let the horizontal axes labeling appear on top.
80       ax.tick_params(top=True, bottom=False,
81                      labeltop=True, labelbottom=False)
82   
83       # Rotate the tick labels and set their alignment.
84       plt.setp(ax.get_xticklabels(), rotation=-30, ha="right",
85                rotation_mode="anchor")
86   
87       # Turn spines off and create white grid.
88       ax.spines[:].set_visible(False)
89   
90       ax.set_xticks(np.arange(data.shape[1]+1)-.5, minor=True)
91       ax.set_yticks(np.arange(data.shape[0]+1)-.5, minor=True)
92       ax.grid(which="minor", color="w", linestyle='-', linewidth=3)
93       ax.tick_params(which="minor", bottom=False, left=False)
94   
95       return im, cbar

97   def annotate_heatmap(im, data=None, valfmt="{x:.2f}",
98                        textcolors=("black", "white"),
99                        threshold=None, **textkw):
100      """ 
101      A function to annotate a heatmap. 
102   
103      Parameters 
104      ---------- 
105      im 
106          The AxesImage to be labeled. 
107      data 
108          Data used to annotate.  If None, the image's data is used.  Optional. 
109      valfmt 
110          The format of the annotations inside the heatmap.  This should either 
111          use the string format method, e.g. "$ {x:.2f}", or be a 
112          `matplotlib.ticker.Formatter`.  Optional. 
113      textcolors 
114          A pair of colors.  The first is used for values below a threshold, 
115          the second for those above.  Optional. 
116      threshold 
117          Value in data units according to which the colors from textcolors are 
118          applied.  If None (the default) uses the middle of the colormap as 
119          separation.  Optional. 
120      **kwargs 
121          All other arguments are forwarded to each call to `text` used to create 
122          the text labels. 
123      """
124  
125      if not isinstance(data, (list, np.ndarray)):
126          data = im.get_array()
127  
128      # Normalize the threshold to the images color range.
129      if threshold is not None:
130          threshold = im.norm(threshold)
131      else:
132          threshold = im.norm(data.max())/2.
133  
134      # Set default alignment to center, but allow it to be
135      # overwritten by textkw.
136      kw = dict(horizontalalignment="center",
137                verticalalignment="center")
138      kw.update(textkw)
139  
140      # Get the formatter in case a string is supplied
141      if isinstance(valfmt, str):
142          valfmt = matplotlib.ticker.StrMethodFormatter(valfmt)
143  
144      # Loop over the data and create a `Text` for each "pixel".
145      # Change the text's color depending on the data.
146      texts = []
147      for i in range(data.shape[0]):
148          for j in range(data.shape[1]):
149              kw.update(color=textcolors[int(im.norm(data[i, j]) > threshold)])
150              text = im.axes.text(j, i, valfmt(data[i, j], None), **kw)
151              texts.append(text)
152  
153      return texts
